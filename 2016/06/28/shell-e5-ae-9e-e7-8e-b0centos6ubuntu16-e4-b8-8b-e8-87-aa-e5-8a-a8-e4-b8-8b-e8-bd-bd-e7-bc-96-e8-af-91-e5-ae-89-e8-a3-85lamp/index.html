目前第一版:
本人在Centos6上运行成功。 
使用方法：先用root账号运行install脚本,然后用普通账号运行init(root账户也可以，不过我没试过).
init 参数说明:
--install-path  安装目录,**当前执行的用户必须有这个目录的权限！！！并且不可使用相对目录！**
比如:

./init --install-path=/home/sxin/soft    这个是对的
./init --install-path=~/soft             这个是错的

init

<pre>`#!/bin/bash
logname='log'
`true &gt; $logname`
echo `date "+%Y-%m-%d %H:%M:%S" `' start' &gt; $logname

isdownphp=true
isdownhttpd=true
isdownmysql=true
isdownnginx=true

downpath=~/downloads
phpcodepath=$downpath/php7
httpdcodepath=$downpath/httpd24/
mysqlcodepath=$downpath/mysql57/
nginxcodepath=$downpath/nginx/
toolscodepath=$downpath/tools
installpath=~/install

for para in $@
do
    [[ "${para}" =~ --php-code-path= ]] &amp;&amp; isdownphp=false &amp;&amp; phpcodepath=${para#--php-code-path=}
    [[ "${para}" =~ --httpd-code-path= ]] &amp;&amp; isdownhttpd=false &amp;&amp; httpdcodepath=${para#--httpd-code-path=}
    [[ "${para}" =~ --mysql-code-path= ]] &amp;&amp; isdownmysql=false &amp;&amp; mysqlcodepath=${para#--mysql-code-path=}
    [[ "${para}" =~ --nginx-code-path= ]] &amp;&amp; isdownnginx=false &amp;&amp; nginxcodepath=${para#--nginx-code-path=}
    [[ "${para}" =~ --install-path= ]] &amp;&amp; installpath=${para#--install-path=}
done

user='sxin'

echo `date "+%Y-%m-%d %H:%M:%S" `' init path' &gt;&gt; $logname

#init path
if [ ! -d "$installpath" ]; then
    mkdir -p "$installpath"
fi
if [ ! -d "$downpath" ]; then
    mkdir -p "$downpath"
fi
if [ ! -d "$phpcodepath" ]; then
    mkdir -p "$phpcodepath"
fi
if [ ! -d "$httpdcodepath" ]; then
    mkdir -p "$httpdcodepath"
fi
if [ ! -d "$mysqlcodepath" ]; then
    mkdir -p "$mysqlcodepath"
fi
if [ ! -d "$nginxcodepath" ]; then
    mkdir -p "$nginxcodepath"
fi
if [ ! -d "$toolscodepath" ]; then
    mkdir -p "$toolscodepath"
fi

echo `date "+%Y-%m-%d %H:%M:%S" `' create path' &gt;&gt; $logname

#下载软件源码
$isdownphp &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' start download php code' &gt;&gt; $logname &amp;&amp; wget -P $downpath/php7 "http://cn2.php.net/distributions/php-7.0.8.tar.gz" &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' download php code ok' &gt;&gt; $logname

$isdownnginx &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' start download nginx code' &gt;&gt; $logname &amp;&amp; wget -P $downpath/nginx "http://nginx.org/download/nginx-1.11.1.tar.gz" &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' download nginx code ok' &gt;&gt; $logname

$isdownhttpd &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' start download httpd code' &gt;&gt; $logname &amp;&amp; wget -P $downpath/httpd24 "http://mirrors.hust.edu.cn/apache//httpd/httpd-2.4.20.tar.gz" &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' download httpd code ok' &gt;&gt; $logname

$isdownmysql &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' start download mysql code' &amp;&amp; wget -P $downpath/mysql57 "http://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-5.7.13.tar.gz" &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' download mysql code ok' &gt;&gt; $logname &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' start download mysql-boost code' &amp;&amp; wget -P $downpath/mysql57 "http://dev.mysql.com/get/Downloads/MySQL-5.7/mysql-boost-5.7.13.tar.gz"&amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' download mysql-boost code ok' &gt;&gt; $logname 

wget -P $toolscodepath "http://apache.fayea.com//apr/apr-1.5.2.tar.gz" "http://apache.fayea.com//apr/apr-util-1.5.4.tar.gz" "http://zlib.net/zlib-1.2.8.tar.gz" "ftp://ftp.csx.cam.ac.uk/pub/software/programming/pcre/pcre-8.39.tar.gz"

echo `date "+%Y-%m-%d %H:%M:%S" `' download apr apr-util zlib pcre code' &gt;&gt; $logname

#extract
tar -zxvf $toolscodepath/apr-1.5.2.tar.gz -C $toolscodepath
echo `date "+%Y-%m-%d %H:%M:%S" `' extract apr success' &gt;&gt; $logname
tar -zxvf $toolscodepath/apr-util-1.5.4.tar.gz -C $toolscodepath
echo `date "+%Y-%m-%d %H:%M:%S" `' extract apr-util success' &gt;&gt; $logname
tar -zxvf $toolscodepath/pcre-8.39.tar.gz -C $toolscodepath
echo `date "+%Y-%m-%d %H:%M:%S" `' extract pcre success' &gt;&gt; $logname
tar -zxvf $toolscodepath/zlib-1.2.8.tar.gz -C $toolscodepath
echo `date "+%Y-%m-%d %H:%M:%S" `' extract zlib success' &gt;&gt; $logname
$isdownnginx &amp;&amp; tar -zxvf $nginxcodepath/nginx-1.11.1.tar.gz -C $nginxcodepath &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' extract nginx success' &gt;&gt; $logname
$isdownphp &amp;&amp; tar -zxvf $phpcodepath/php-7.0.8.tar.gz -C $phpcodepath &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' extract php success' &gt;&gt; $logname
$isdownmysql &amp;&amp; tar -zxvf $mysqlcodepath/mysql-5.7.13.tar.gz -C $mysqlcodepath &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' extract mysql success' &gt;&gt; $logname
$isdownmysql &amp;&amp; tar -zxvf $mysqlcodepath/mysql-boost-5.7.13.tar.gz -C $mysqlcodepath &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' extract mysql-boost success' &gt;&gt; $logname
$isdownhttpd &amp;&amp; tar -zxvf $httpdcodepath/httpd-2.4.20.tar.gz -C $httpdcodepath &amp;&amp; echo `date "+%Y-%m-%d %H:%M:%S" `' extract httpd success' &gt;&gt; $logname

#compile
#apr
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile apr ' &gt;&gt; $logname
cd $toolscodepath/apr-1.5.2 &amp;&amp; ./configure --prefix=$installpath/tools/apr &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `' compile install apr success' &gt;&gt; $logname
#apr-util
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile apr-util ' &gt;&gt; $logname
cd $toolscodepath/apr-util-1.5.4 &amp;&amp; ./configure --prefix=$installpath/tools/apr-util --with-apr=$installpath/tools/apr &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `'  apr-util success' &gt;&gt; $logname
#pcre
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile pcre' &gt;&gt; $logname
cd $toolscodepath/pcre-8.39 &amp;&amp; ./configure --prefix=$installpath/tools/pcre &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `'  pcre success' &gt;&gt; $logname
#zlib
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile zlib' &gt;&gt; $logname
cd $toolscodepath/zlib-1.2.8 &amp;&amp; ./configure --prefix=$installpath/tools/zlib &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `'  zlib success' &gt;&gt; $logname
#nginx
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile nginx' &gt;&gt; $logname
cd $nginxcodepath/nginx-1.11.1 &amp;&amp; ./configure --prefix=$installpath/nginx --with-pcre=$downpath/tools/pcre-8.39 --with-zlib=$downpath/tools/zlib-1.2.8 &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `'  nginx success' &gt;&gt; $logname
#httpd
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile httpd' &gt;&gt; $logname
cd $httpdcodepath/httpd-2.4.20 &amp;&amp; ./configure --prefix=$installpath/httpd24 --with-apr=$installpath/tools/apr --with-apr-util=$installpath/tools/apr-util/ --with-pcre=$installpath/tools/pcre/ &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `'  httpd success' &gt;&gt; $logname
#mysql
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile mysql' &gt;&gt; $logname
ncuresespath=`find /usr -name libncurses.so`
cd $mysqlcodepath/mysql-5.7.13 &amp;&amp; cmake . -DCMAKE_INSTALL_PREFIX=$installpath/mysql57 -DSYSCONFDIR=/etc/mysql -DDEFAULT_CHARSET=utf8 -DDEFAULT_COLLATION=utf8_general_ci -DMYSQL_DATADIR=$installpath/mysql/data -DWITH_BOOST=boost -DCURSES_LIBRARY=$ncuresespath -DCURSES_INCLUDE_PATH=/usr/include &amp;&amp; make &amp;&amp; make install
echo `date "+%Y-%m-%d %H:%M:%S" `'  mysql success' &gt;&gt; $logname
#$installpath/mysql57/bin/mysqld --initialize --user=$user --basedir=$installpath/mysql --datadir=$installpath/mysql/data
echo `date "+%Y-%m-%d %H:%M:%S" `'  init mysql data success' &gt;&gt; $logname
#php
`sed -i "/PDO_MYSQL configure failed, MySQL 4.1 needed/"d $phpcodepath/php-7.0.8/configure`
echo `date "+%Y-%m-%d %H:%M:%S" `' start compile php' &gt;&gt; $logname
cd $phpcodepath/php-7.0.8 &amp;&amp; ./configure --prefix=$installpath/php7 --with-config-file-path=$installpath/php7/conf --with-apxs2=$installpath/httpd24/bin/apxs --with-pdo-mysql=$installpath/mysql57 --with-mysqli --with-zlib-dir=$installpath/tools/zlib &amp;&amp; make &amp;&amp; make install
mkdir $installpath/php7/conf
cp $phpcodepath/php-7.0.8/php.ini-development $installpath/php7/conf/php.ini
echo `date "+%Y-%m-%d %H:%M:%S" `'  php success' &gt;&gt; $logname
