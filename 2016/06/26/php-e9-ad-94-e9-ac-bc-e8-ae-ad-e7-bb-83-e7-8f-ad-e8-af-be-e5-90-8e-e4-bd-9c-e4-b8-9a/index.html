godinit

<pre>`&lt;?php
    class godinit
    {
        static $VERSION="god version is 1.1";
        static $prj_name="";//项目名称
        static $prj_author="sxin";//项目作者
        static function init()
        {
            echo "input your project name?".PHP_EOL;
            self::$prj_name=trim(fgets(STDIN));
            echo "input your project author?".PHP_EOL;
            self::$prj_author=trim(fgets(STDIN));

            return initConfig(self::$prj_name,self::$prj_author);
        }

        static function make()
        {
            $pchar=new Phar("god.phar");
            $pchar-&gt;buildFromDirectory(dirname(__FILE_));
            $pchar-&gt;setStub($pchar-&gt;createDefaultStub('god'));
            $pchar-&gt;compressFiles(Phar::GZ);
        }
    }
`</pre>
